## Задача 01
<h3>
Да се намери резултатът от изпълнението на следната програма:

```C++
#include<iostream>

class Q {
    public:
        void f();
        void g();
        int x;
};

void Q::f() {
    std::cout << "f()\n" << "x = " << std:: endl; 
}

void Q::g() {
    std::cout << "g()\n" << "x = " << std:: endl; 
}

int main() {
    Q obj;

    int Q::*px = &Q::x;
    obj.*px = 25;
    std::cout << obj.*px << std::endl;

    void (Q::*ptr)() = &Q::f;
    (obj.*ptr)();

    ptr = &Q::g;
    (obj.*ptr)();

    return 0;
}

```
